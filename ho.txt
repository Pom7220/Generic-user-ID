async function init(){

async function loadAll(){
  setSyncState('syncing');
  // Load all BUs
  const {data:bus}=await sb.from('business_units').select('*').order('name');
  ALL_BUS=bus||[];

  // Load all stores
  const {data:stores}=await sb.from('stores').select('*').eq('active',true).order('code');
  ALL_STORES=stores||[];

  // Load all apps
  const {data:apps}=await sb.from('apps').select('*').eq('active',true).order('code');
  ALL_APPS=apps||[];

  // Load all employees
  const {data:emps}=await sb.from('employees').select('id,emp_code,full_name,status,bu_id,store_id').order('full_name');
  ALL_EMPS=emps||[];

  // Load all assignments with app info
  const storeIds=ALL_STORES.map(s=>s.id);
  if(storeIds.length){
    const {data:asn}=await sb.from('generic_id_assignments')
      .select('id,store_id,app_id,generic_id,status,pw_last_rotated,app:app_id(code,name,pw_rotation_days)')
      .in('store_id',storeIds);
    ALL_ASSIGNMENTS=asn||[];
  } else {ALL_ASSIGNMENTS=[];}

  // Pre-compute per-BU data
  ALL_BUS.forEach((bu,i)=>{
    const stores=ALL_STORES.filter(s=>s.bu_id===bu.id);
    const stIds=stores.map(s=>s.id);
    const asn=ALL_ASSIGNMENTS.filter(a=>stIds.includes(a.store_id));
    const emps=ALL_EMPS.filter(e=>e.bu_id===bu.id);
    const apps=ALL_APPS.filter(a=>a.bu_id===bu.id);
    const pwDue=asn.filter(a=>a.status==='active'&&isPwOverdue(a)).length;
    const risk=asn.filter(a=>['at_risk','orphan'].includes(a.status)).length;
    const riskScore=Math.min(100,Math.round((pwDue*3+risk*10)));
    BU_DATA[bu.id]={bu,stores,asn,emps,apps,pwDue,risk,riskScore,gradient:BU_GRADIENTS[i%BU_GRADIENTS.length]};
  });

  renderAll();
  setSyncState('ok');
}

function renderAll(){
  renderMetrics();renderBUCards();renderBUFull();renderRisk();renderApps();renderSystemStats();
}

function setSyncState(s){
  const p=document.getElementById('sync-pill'),l=document.getElementById('sync-label');
  p.className='sync-pill'+(s==='syncing'?' syncing':'');
  l.textContent=s==='ok'?'Live':s==='syncing'?'Syncing…':'Offline';
}

// ── METRICS ───────────────────────────────────────────────────────
function renderMetrics(){
  const totalIds=ALL_ASSIGNMENTS.length;
  const active=ALL_ASSIGNMENTS.filter(a=>a.status==='active').length;
  const pwDue=ALL_ASSIGNMENTS.filter(a=>a.status==='active'&&isPwOverdue(a)).length;

    {l:'BUs',v:ALL_BUS.length,c:'purple'},
    {l:'Stores',v:ALL_STORES.length,c:'blue'},

    {l:'Pw overdue',v:pwDue,c:pwDue?'amber':'green'},
    {l:'Risk alerts',v:risk,c:risk?'red':'green'},
  ];
  document.getElementById('ho-metrics').innerHTML=metrics.map(m=>`
    <div class="mcard"><div class="mcard-label">${m.l}</div><div class="mcard-value ${m.c}">${m.v}</div></div>`).join('');

          <div class="risk-bar"><div class="risk-fill" style="width:${d.riskScore}%;background:${riskColor}"></div></div>

        </div>
      </div>
      <span style="color:var(--hint);font-size:20px">›</span>
    </div>
    <div class="bu-card-stats">
      <span class="badge b-info">${d.asn.length} IDs</span>

// ── RISK ──────────────────────────────────────────────────────────
function renderRisk(){
  const allRisk=[];
  ALL_BUS.forEach(bu=>{
    const d=BU_DATA[bu.id]||{};
    const atRisk=d.asn?.filter(a=>a.status==='at_risk')||[];
    const orphans=d.asn?.filter(a=>a.status==='orphan')||[];
    const pwOver=d.asn?.filter(a=>a.status==='active'&&isPwOverdue(a))||[];
    if(atRisk.length) allRisk.push({type:'danger',icon:'⚠️',bu:bu.name,title:`${atRisk.length} at-risk ID(s)`,desc:`Supervisor resigned. IDs: ${atRisk.map(a=>a.generic_id).join(', ')}`});
    if(orphans.length) allRisk.push({type:'danger',icon:'⛔',bu:bu.name,title:`${orphans.length} orphan session(s)`,desc:`Login with no assignment: ${orphans.map(a=>a.generic_id).join(', ')}`});
    if(pwOver.length) allRisk.push({type:'warn',icon:'🔑',bu:bu.name,title:`${pwOver.length} password(s) overdue`,desc:`IDs: ${pwOver.map(a=>a.generic_id).join(', ')}`});
  });

  const html=allRisk.length
    ?allRisk.map(r=>`<div class="risk-card ${r.type}">
        <div class="risk-icon">${r.icon}</div>
        <div><div style="font-size:11px;font-weight:600;opacity:.7;margin-bottom:3px">${r.bu}</div>
        <div class="risk-title">${r.title}</div><div class="risk-desc">${r.desc}</div></div>
      </div>`).join('')
    :'<div class="empty"><div class="empty-icon">✅</div>No open risks across all BUs</div>';

  document.getElementById('ho-risk-summary').innerHTML=allRisk.slice(0,3).map(r=>`<div class="risk-card ${r.type}">
    <div class="risk-icon">${r.icon}</div>
    <div><div style="font-size:11px;font-weight:600;opacity:.7;margin-bottom:2px">${r.bu}</div>
    <div class="risk-title">${r.title}</div></div></div>`).join('')||'<div class="empty"><div class="empty-icon">✅</div>No open risks</div>';
  document.getElementById('ho-risk-full').innerHTML=html;
}

// ── APPS ──────────────────────────────────────────────────────────
function renderApps(){

  if(!appCodes.length){el.innerHTML='<div class="empty"><div class="empty-icon">📱</div>No apps registered</div>';return;}

  el.innerHTML=appCodes.map(code=>{
    const appsForCode=ALL_APPS.filter(a=>a.code===code);
    const totalIds=ALL_ASSIGNMENTS.filter(a=>a.app?.code===code).length;
    const active=ALL_ASSIGNMENTS.filter(a=>a.app?.code===code&&a.status==='active').length;
    const pwDue=ALL_ASSIGNMENTS.filter(a=>a.app?.code===code&&a.status==='active'&&isPwOverdue(a)).length;
    const color=APP_COLORS[code]||APP_COLORS.default;
    const appName=appsForCode[0]?.name||code;
    return `<div class="gcard" style="margin-bottom:12px">
      <div class="gcard-header">
        <div style="display:flex;align-items:center;gap:10px">
          <div style="width:42px;height:42px;border-radius:12px;background:${color}22;display:flex;align-items:center;justify-content:center;font-size:15px;font-weight:800;color:${color}">${code}</div>
          <div><div style="font-size:15px;font-weight:700">${appName}</div>
          <div style="font-size:12px;color:var(--muted)">${appsForCode.length} BU(s) · avg ${appsForCode[0]?.pw_rotation_days||30}d rotation</div></div>
        </div>
        <div style="display:flex;gap:6px;flex-wrap:wrap">
          <span class="badge b-info">${totalIds} IDs</span>
          <span class="badge b-ok">${active} active</span>
          ${pwDue?`<span class="badge b-warn">${pwDue} pw due</span>`:''}
        </div>
      </div>
      <div style="padding:12px 16px;display:flex;gap:7px;flex-wrap:wrap">
        ${ALL_BUS.map(bu=>{
          const buIds=ALL_ASSIGNMENTS.filter(a=>a.app?.code===code&&BU_DATA[bu.id]?.asn?.includes(a)).length;
          const stIds=(BU_DATA[bu.id]?.stores||[]).map(s=>s.id);
          const c=ALL_ASSIGNMENTS.filter(a=>a.app?.code===code&&stIds.includes(a.store_id)).length;
          return `<span class="badge b-neutral">${bu.code}: ${c}</span>`;
        }).join('')}
      </div>
    </div>`;
  }).join('');

async function saveBU(){
  const code=document.getElementById('bu-code').value.trim().toUpperCase();
  const name=document.getElementById('bu-name').value.trim();
  const errEl=document.getElementById('bu-err');
  if(!code||!name){errEl.textContent='Code and name required.';errEl.classList.remove('hidden');return;}
  const {error}=await sb.from('business_units').insert({code,name});
  if(error){errEl.textContent=error.message;errEl.classList.remove('hidden');return;}
  closeSheet('sheet-bu');await loadAll();
}

// ── EXPORT ────────────────────────────────────────────────────────
function exportReport(){
  const rows=[['BU','Stores','Employees','Total IDs','Active IDs','Pw Overdue','Risk','Risk Score']];
  ALL_BUS.forEach(bu=>{
    const d=BU_DATA[bu.id]||{};
    rows.push([bu.name,d.stores?.length||0,d.emps?.filter(e=>e.status==='active').length||0,
      d.asn?.length||0,d.asn?.filter(a=>a.status==='active').length||0,
      d.pwDue||0,d.risk||0,d.riskScore||0]);
  });
  const csv=rows.map(r=>r.join(',')).join('\n');
  const a=document.createElement('a');
  a.href='data:text/csv;charset=utf-8,'+encodeURIComponent(csv);
  a.download=`ho_report_${new Date().toISOString().slice(0,10)}.csv`;
  a.click();
}

function isPwOverdue(a){if(!a.pw_last_rotated)return false;return Math.floor((Date.now()-new Date(a.pw_last_rotated))/(864e5))>=(a.app?.pw_rotation_days||30)}
function gotoBU(buId){
  // If specific BU requested (from card click), go directly
  if(buId){
    const bu=ALL_BUS.find(b=>b.id===buId)||ALL_BUS[0];
    sessionStorage.setItem('gid_bu_id',bu.id);
    sessionStorage.setItem('gid_bu_name',bu.name);
    sessionStorage.setItem('gid_role','ho_admin');
    window.location.href='bu.html';
    return;
  }
  // Multiple BUs — show picker sheet so user chooses explicitly
  if(ALL_BUS.length>1){
    document.getElementById('bu-picker-list').innerHTML=ALL_BUS.map((bu,i)=>`
      <div onclick="gotoBU('${bu.id}')" style="display:flex;align-items:center;gap:14px;padding:14px 16px;cursor:pointer;border-bottom:0.5px solid var(--border2)">
        <div style="width:42px;height:42px;border-radius:12px;background:${['linear-gradient(135deg,#0A84FF,#5AC8FA)','linear-gradient(135deg,#30D158,#34C759)','linear-gradient(135deg,#FF9F0A,#FF6B35)','linear-gradient(135deg,#BF5AF2,#9B59B6)'][i%4]};display:flex;align-items:center;justify-content:center;font-size:14px;font-weight:800;color:#fff;flex-shrink:0">${bu.code.slice(0,2)}</div>
        <div><div style="font-size:15px;font-weight:600;color:var(--text)">${bu.name}</div><div style="font-size:12px;color:var(--muted)">HO Admin · All access</div></div>
        <span style="margin-left:auto;color:var(--hint);font-size:20px">›</span>
      </div>`).join('');
    openSheet('sheet-bu-picker');
    return;
  }
  // Single BU — go directly
  const bu=ALL_BUS[0];
  if(!bu) return;
  sessionStorage.setItem('gid_bu_id',bu.id);
  sessionStorage.setItem('gid_bu_name',bu.name);
  sessionStorage.setItem('gid_role','ho_admin');
  window.location.href='bu.html';
}
